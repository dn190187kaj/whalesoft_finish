Задача №2. Релизовать функцию createList(listData, listContainer, itemContainer), возвращаюшую узел списка. Второй и третий аргументы не обязательные. Значения по умолчанию для них - ul и li. listData - массив. listData Может содержать как элементы (текст), так и массивы элементов. Глубина вложенности массивов любая. 

function createList(listData, listContainer = 'ul', itemContainer = 'li', result){
		var result = result||['<' + listContainer + '>'];

		for (var i = 0; i < listData.length; i++) {
			if (Array.isArray(listData[i])) {
				result.push('<' + listContainer + '>')
				createList(listData[i], listContainer, itemContainer, result);
				result.push('')
			} else {
				var teg = '<' + itemContainer + '>' + listData[i] + ''
				result.push(teg);
			}

		}

			result = result.join('');
			result += '';
			return $('#result').html(result);
	}
		
createList(['мясо', ['яблоки', [[['клубника']]],'бананы'], 'орехи'], 'ol');
